long long qpow(long long a, long long b, long long mod) {
    long long res = 1;
    while (b) {
        if (b & 1) {
            res = res * a % mod;
        }
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}
int main() {

    cin.tie(0);
    cout.tie(0);
    int n, m;

    cout << qpow(10, m, 10 * n) / n;
    return 0;

long long f[18];

long long strll;
long long get(int index, long long mid) {
    mid--;
    if (index + str.size() >= 18) {
        return 0x7fffffffffffffff;
    }
    if (str[0] == '0') {
        mid += f[index];
    }
    long long p = mid % f[index];
    long long q = mid / f[index];
    return q * f[index + str.size()] + strll * f[index] + p;
}
void solve() {
    cin >> str;
    strll = 0;
    for (auto& i : str) {
        strll = strll * 10 + i - '0';
    }
    auto get_sum = [&](long long x) {
        long long ans = 0;
        for (int i = 0 ; i < 16 ; i++) {
            long long L = 0, R = f[16 - str.size()];
            while (L < R) {
                long long mid = L + R + 1 >> 1;
                if (get(i, mid) <= x) {
                    L = mid;
                } else {
                    R = mid - 1;
                }
            }
            ans += L;
        }
        return ans;
    };
    long long L, R;
    cin >> L >> R;
    cout << get_sum(R) - get_sum(L - 1) << "\n";
}
int main() {
    ios::sync_with_stdio(0);

    f[0] = 1;
    for (int i = 1 ; i < 18 ; i++) {
        f[i] = f[i - 1] * 10;
    }

        solve();
    }
    return 0;
}

int lowbit(int x){
    return (x)&(-x);

    while (x <= n + 1) {
        tree[x] += v;
        x += lowbit(x);

    int res = 0;
    while (x) {
        res += tree[x];
        x -= lowbit(x);
    }

int main(){
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);

    vector<long long> sum(n + 1);
    vector<int> sumXor(n + 1);
    tree.resize(n + 2);
    for(int i = 1 ; i <= n ; i++){
        cin >> a[i];
        sumXor[i] = sumXor[i - 1] ^ a[i];
        sum[i] = sum[i - 1] + a[i];
    }
    long long ans1 = 0, ans2 = 0;
    for (int i = 20 ; i >= 0 ; i--) {
        int cnt[2] = {0, 0};
        for (int j = 0 ; j <= n ; j++) {
            int k = sumXor[j] >> i & 1;
            ans1 = (ans1 + cnt[k ^ 1] * (1LL << i) % mod) % mod;
            cnt[k]++;
        }
    }
    vector<long long> p(n + 1);
    auto get = [&](long long x) -> int {
        return upper_bound(begin(p), end(p), x) - begin(p);
    };
    for (int i = 35 ; i >= 0 ; i--){
        long long res = 0;
        for(int j = 0 ; j <= n ; j++) {
            p[j] = sum[j] & ((1LL << i + 1) - 1);
        }
        sort(begin(p), end(p));
        fill(begin(tree), end(tree), 0);
        for (int j = 0 ; j <= n ; j++) {
            long long k = sum[j] & ((1LL << i + 1) - 1);
            add(get(k), 1);
            int t1 = query(get(k - (1LL << i)));
            int t2 = query(get(k + (1LL << i)));
            int t3 = query(get(k));
            res += t1 + t2 - t3;
        }
        if (res & 1) {
            ans2 |= 1LL << i;
        }
    }
    cout << ans1 << " " << ans2 << "\n";
    return 0;
}

long long euler_phi(long long n) {
    long long ans = n;
    for (long long i = 2 ; i * i <= n ; i++) {
        if (n % i == 0) {
            ans = ans / i * (i - 1);
            while (n % i == 0) {
                n /= i;
            }
        }
    }
    if (n > 1) {
        ans = ans / n * (n - 1);
    }
    return ans;
}

long long qpow(long long a, long long b, long long mod) {
    long long res = 1;
    while (b) {
        if (b & 1) {
            res = (__int128) res * a % mod;
        }
        a = (__int128) a * a % mod;
        b >>= 1;
    }
    return res;
}

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    long long p, q;
    cin >> p >> q;
    long long k = __gcd(p, q);
    p /= k;
    q /= k;
    int cnt2 = 0, cnt5 = 0;
    while (q % 2 == 0) {
        q /= 2;
        cnt2++;
    }
    while (q % 5 == 0) {
        q /= 5;
        cnt5++;
    }
    if (q == 1) {
        cout << -1;
    } else {
        cout << max(cnt2, cnt5) << " ";
        long long ans = 0x7fffffffffffffff;
        long long phi = euler_phi(q);
        for (long long i = 1 ; i * i <= phi ; i++) {
            if (phi % i == 0) {
                if (qpow(10, i, q) == 1) {
                    ans = min(ans, i);
                }
                if (qpow(10, phi / i, q) == 1) {
                    ans = min(ans, phi / i);
                }
            }
        }
        cout << ans << "\n";
    }
    return 0;
}

    #include <bits/stdc++.h>
using namespace std;
// 2023 OneWan
const int N = 1000005;
int a[N];
struct FenwickTree {
    int mx;
    vector<long long> sum;
    FenwickTree(int N) {
        mx = N;
        sum.resize(N + 1);
    }
    static int lowbit(int x) {
        return x & -x;
    }
    void add(int x, int k) {
        while (x <= mx) {
            sum[x] += k;
            x += lowbit(x);
        }
    }
    long long query(int x) {
        long long res = 0;
        while (x) {
            res += sum[x];
            x -= lowbit(x);
        }
        return res;
    }
};
struct Block {
    vector<int> st, ed, bel, sz, pos;
    vector<bool> exist;
    vector<FenwickTree> ft;
    int size;
    long long ans;
    Block(int n) {
        ans = 0;
        int m = sqrt(n);
        int k = n / m + 1;
        ft.resize(k + 1, FenwickTree(n + 1));
        st.resize(k + 1);
        ed.resize(k + 1);
        sz.resize(k + 1);
        exist.resize(n + 1, false);
        bel.resize(n + 1);
        pos.resize(n + 1);
        int cnt = 0;
        size = 1;
        st[1] = 1;
        for (int i = 1 ; i <= n ; i++) {
            pos[a[i]] = i;
            if (cnt == m) {
                cnt = 0;
                size++;
                st[size] = i;
                ed[size - 1] = i - 1;
            }
            bel[i] = size;
            cnt++;
            sz[size]++;
        }
        ed[size] = n;
        for (int i = 1 ; i <= size ; i++) {
            for (int j = st[i] ; j <= ed[i] ; j++) {
                ft[i].add(a[j], 1);
            }
        }
        FenwickTree fft(n + 1);
        for (int i = n ; i >= 1 ; i--) {
            ans += fft.query(a[i]);
            fft.add(a[i], 1);
        }
    }
    void remove(int x) {
        int now = pos[x], be = bel[now];
        for (int i = st[be] ; i < now ; i++) {
            if (exist[i]) continue;
            if (a[i] > x) ans--;
        }
        for (int i = now + 1 ; i <= ed[be] ; i++) {
            if (exist[i]) continue;
            if (a[i] < x) ans--;
        }
        exist[now] = true;
        for (int i = 1 ; i < be ; i++) {
            ans -= sz[i] - ft[i].query(x);
        }

        for (int i = be + 1 ; i <= size ; i++) {
            ans -= ft[i].query(x);
        }
        ft[be].add(x, -1);
        sz[be]--;
    }
    long long query() {
        return ans;
    }
};
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, m;
    cin >> n >> m;
    for (int i = 1 ; i <= n ; i++) {
        cin >> a[i];
    }
    Block block(n);
    for (int i = 1 ; i <= m ; i++) {
        int x;
        cin >> x;
        cout << block.query() << "\n";
        block.remove(x);
    }
    return 0;
}

int belong[100005];
struct MQuery {
    int L, R, idx;
    bool operator<(const MQuery& x) const {
        if (belong[L] != belong[x.L]) return L < x.L;
        if (belong[L] & 1) return R < x.R;
        return R > x.R;
    }
};
vector<int> c(1);
bitset<100001> bs[100000 / 3 + 1], res;
int cnt[100005];
int a[100005];
bool vis[100005];
int n;
void add(int x) {
    res[cnt[x] + x] = 1;
    cnt[x]++;
}
void del(int x) {
    cnt[x]--;
    res[cnt[x] + x] = 0;
}
void solve(int q) {
    vector<MQuery> mq;
    vector<int> ans(q);
    for (int i = 0 ; i < q ; i++) {
        for (int j = 0 ; j < 3 ; j++) {
            int L, R;
            cin >> L >> R;
            mq.emplace_back(L, R, i);
            ans[i] += R - L + 1;
        }
    }
    sort(begin(mq), end(mq));
    fill(vis, vis + n + 1, false);
    fill(cnt, cnt + n + 1, 0);
    res.reset();
    for (int i = 0, L = 1, R = 0 ; i < 3 * q ; i++) {
        while (L > mq[i].L) add(a[--L]);
        while (R < mq[i].R) add(a[++R]);
        while (L < mq[i].L) del(a[L++]);
        while (R > mq[i].R) del(a[R--]);
        if (vis[mq[i].idx]) bs[mq[i].idx] &= res;
        else {
            vis[mq[i].idx] = true;
            bs[mq[i].idx] = res;
        }
    }
    for (int i = 0 ; i < q ; i++) {
        ans[i] -= bs[i].count() * 3;
        cout << ans[i] << "\n";
    }
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int q;
    cin >> n >> q;
    int mx = sqrt(n);
    for (int i = 1 ; i <= n ; i++) {
        belong[i] = i / mx;
    }
    for (int i = 1 ; i <= n ; i++) {
        cin >> a[i];
        c.emplace_back(a[i]);
    }
    sort(begin(c), end(c));
    for (int i = 1 ; i <= n ; i++) {
        a[i] = lower_bound(begin(c), end(c), a[i]) - begin(c);
    }
    int B = q / 3;
    solve(B);
    solve(B);
    solve(q - 2 * B);
    return 0;
}

// 2023 OneWan

int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n;
    cin >> n;
    vector<int> p(n + 1);
    for (int i = 1 ; i <= n ; i++) {
        cin >> p[i];
    }
    vector<int> min_L(n + 2, 0), min_R(n + 2, n + 1), max_L(n + 2, 0), max_R(n + 2, n + 1);
    stack<int> smin, smax;
    for (int i = 1 ; i <= n ; i++) {
        while (!smin.empty() && p[i] < p[smin.top()]) {
            min_R[smin.top()] = i;
            smin.pop();
        }
        if (!smin.empty()) {
            min_L[i] = smin.top();
        }
        smin.emplace(i);
        while (!smax.empty() && p[i] > p[smax.top()]) {
            max_R[smax.top()] = i;
            smax.pop();
        }
        if (!smax.empty()) {
            max_L[i] = smax.top();
        }
        smax.emplace(i);
    }
    long long ans = 0;
    for (int i = 1 ; i <= n ; i++) {
        int L = max_L[i], R = max_R[i];
        if (R - i < i - L) {
            int now = n;
            for (int j = i ; j < R ; j++) {
                now = min(now, min_L[j]);
                ans += max(0, now - L);
            }
        } else {
            int now = i;
            for (int j = i ; j > L ; j--) {
                if (p[j] < p[now]) {
                    now = j;
                }
                if (now != i) {
                    ans += max(0, min(min_R[now], R) - i);
                }
            }
        }
    }
    cout << ans << "\n";
    return 0;
}
