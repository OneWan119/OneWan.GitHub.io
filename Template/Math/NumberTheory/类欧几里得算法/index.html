    类欧几里得算法

    \(f(a,b,c,n)=\sum_{i=0}^n\lfloor\frac{a\cdot{i}+b}{c}\rfloor\)

    当 \(a\ge{c}\text{ 或 }b\ge{c}\)

    \[ \begin{align} f(a,b,c,n)&=\sum_{i=0}^n\lfloor\dfrac{a\cdot{i}+b}{c}\rfloor\\ &=\sum_{i=0}^n\left\lfloor\frac{\left(\lfloor\frac{a}{c}\rfloor\cdot{c}+a\ mod\ c\right)\cdot{i}+\left(\lfloor\frac{b}{c}\rfloor\cdot{c}+b\ mod\ c\right)}{c}\right\rfloor\\ &=\sum_{i=0}^n\left\lfloor\lfloor\frac{a}{c}\rfloor\cdot{i}+\frac{\left(a\ mod\ c\right)\cdot{i}}{c}+\lfloor\frac{b}{c}\rfloor+\frac{\left(b\ mod\ c\right)}{c}\right\rfloor\\ &=\frac{n\cdot(n+1)}{2}\cdot\lfloor\frac{a}{c}\rfloor+(n+1)\cdot\lfloor{\frac{b}{c}}\rfloor+\sum_{i=0}^n\left\lfloor\frac{(a\ mod\ c)\cdot{i}+(b\ mod\ c)}{c}\right\rfloor\\ &=\frac{n\cdot(n+1)}{2}\cdot\lfloor\frac{a}{c}\rfloor+(n+1)\cdot\lfloor{\frac{b}{c}}\rfloor+f(a\ mod\ c,b\ mod\ c,c,n) \end{align} \]

    当 \(a<c\text{ 且 }b<c\)

    \[ \begin{align} f(a,b,c,n)&=\sum_{i=0}^n\lfloor\dfrac{a\cdot{i}+b}{c}\rfloor\\ &=\sum_{i=0}^n\sum_{j=0}^{\lfloor\frac{a\cdot{i}+b}{c}\rfloor-1}1\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}\sum_{i=0}^n\left[j<\lfloor\frac{a\cdot{i}+b}{c}\rfloor\right] \end{align} \]
    \[ \begin{align} j<\lfloor\frac{a\cdot{i}+b}{c}\rfloor\Longleftrightarrow{j}+1\leq\frac{a\cdot{i}+b}{c}\Longleftrightarrow{j}\cdot{c}+c\le{a}\cdot{i}+b\Longleftrightarrow\left\lfloor\frac{j\cdot{c}+c-b-1}{a}\right\rfloor<i \end{align} \]
    \[ \begin{align} f(a,b,c,b)&=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}\sum_{i=0}^n\left[\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor<i\right]\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}\left(n-\left\lfloor\frac{j\cdot{c}+c-b-1}{a}\right\rfloor\right)\\ &=n\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor-f(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1) \end{align} \]

    \(g(a,b,c,n)=\sum_{i=0}^ni\cdot\lfloor\frac{a\cdot{i}+b}{c}\rfloor\)

    当 \(a\ge{c}\text{ 或 }b\ge{c}\)

    \[ \begin{align} g(a,b,c,n)&=\sum_{i=0}^ni\cdot\lfloor\frac{a\cdot{i}+b}{c}\rfloor\\ &=\frac{n\cdot(n+1)\cdot(2n+1)}{6}\cdot\lfloor\frac{a}{c}\rfloor+\frac{n\cdot(n+1)}{2}\cdot\lfloor\frac{b}{c}\rfloor+g(a\ mod\ c,b\ mod\ c,c,n) \end{align} \]

    当 \(a<c\text{ 且 }b<c\)

    \[ \begin{align} g(a,b,c,n)&=\sum_{i=0}^ni\cdot\lfloor\dfrac{a\cdot{i}+b}{c}\rfloor\\ &=\sum_{i=0}^n\sum_{j=0}^{\lfloor\frac{a\cdot{i}+b}{c}\rfloor-1}i\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}\sum_{i=0}^ni\cdot\left[j<\lfloor\frac{a\cdot{i}+b}{c}\rfloor\right]\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}\sum_{i=0}^ni\cdot\left[\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor<i\right]\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}\frac{(\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor+1+n)\cdot(n-\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor)}{2}\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}\left(\frac{n^2+n-\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor-(\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor)^2}{2}\right)\\ &=\frac{1}{2}\cdot{n}\cdot(n+1)\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor-\frac{1}{2}\cdot{h}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)\\ &-\frac{1}{2}\cdot{f}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1) \end{align} \]

    \(h(a,b,c,n)=\sum_{i=0}^n\left\lfloor\frac{a\cdot{i}+b}{c}\right\rfloor^2\)

    当 \(a\ge{c}\text{ 或 }b\ge{c}\)

    \[ \begin{align} h(a,b,c,n)&=\sum_{i=0}^n\left\lfloor\frac{a\cdot{i}+b}{c}\right\rfloor^2\\ &=(n+1)\cdot\lfloor\frac{b}{c}\rfloor^2+n\cdot(n+1)\cdot\lfloor\frac{a}{c}\rfloor\cdot\lfloor\frac{b}{c}\rfloor+\frac{n\cdot(n+1)\cdot(2n+1)}{6}\cdot\lfloor\frac{a}{c}\rfloor^2\\ &+2\cdot\lfloor\frac{b}{c}\rfloor\cdot{f}(a\ mod\ c,b\ mod\ c,c,n) +2\cdot\lfloor\frac{a}{c}\rfloor\cdot{g}(a\ mod\ c,b\ mod\ c,c,n) \end{align} \]

    当 \(a<c\text{ 且 }b<c\)

    \[ \begin{align} n^2&=2\cdot\frac{n\cdot(n+1)}{2}-n=2\cdot(\sum_{i=0}^ni)-n\\ h(a,b,c,n)&=\sum_{i=0}^n\left\lfloor\frac{a\cdot{i}+b}{c}\right\rfloor^2\\ &=\sum_{i=0}^n\left[2\cdot(\sum_{j=1}^{\lfloor\frac{a\cdot{i}+b}{c}\rfloor}j)-\lfloor\frac{a\cdot{i}+b}{c}\rfloor\right]\\ &=2\cdot(\sum_{i=0}^n\sum_{j=1}^{\lfloor\frac{a\cdot{i}+b}{c}\rfloor}j)-f(a,b,c,n)\\ \sum_{i=0}^n\sum_{j=1}^{\lfloor\frac{a\cdot{i}+b}{c}\rfloor}j&=\sum_{i=0}^n\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}j+1\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}(j+1)\sum_{i=0}^n\left[j<\lfloor\frac{a\cdot{i}+b}{c}\rfloor\right]\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}(j+1)\sum_{i=0}^n[\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor<i]\\ &=\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}(j+1)\cdot(n-\lfloor\frac{j\cdot{c}+c-b-1}{a}\rfloor)\\ &=\frac{1}{2}\cdot{n}\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor\cdot(\lfloor\frac{a\cdot{n}+b}{c}\rfloor+1)-\sum_{j=0}^{\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1}(j+1)\cdot\left\lfloor\frac{j\cdot{c}+c-b-1}{a}\right\rfloor\\ &=\frac{1}{2}\cdot{n}\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor\cdot(\lfloor\frac{a\cdot{n}+b}{c}\rfloor+1)-g(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)\\ &-f(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)\\ h(a,b,c,n)&=n\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor\cdot(\lfloor\frac{a\cdot{n}+b}{c}\rfloor+1)-2\cdot{g}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)\\ &-2\cdot{f}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)-f(a,b,c,n)\\ \end{align} \]

    总结

    \[ f(a,b,c,n)=\sum_{i=0}^n\lfloor\frac{a\cdot{i}+b}{c}\rfloor= \begin{cases} \frac{n\cdot(n+1)}{2}\cdot\lfloor\frac{a}{c}\rfloor+(n+1)\cdot\lfloor{\frac{b}{c}}\rfloor+f(a\ mod\ c,b\ mod\ c,c,n)&a\ge{c}\text{ 或 }b\ge{c}\\ n\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor-f(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)&a<c\text{ 且 }b<c\\ \end{cases} \]
    \[ g(a,b,c,n)=\sum_{i=0}^ni\cdot\lfloor\frac{a\cdot{i}+b}{c}\rfloor= \begin{cases} \frac{n\cdot(n+1)\cdot(2n+1)}{6}\cdot\lfloor\frac{a}{c}\rfloor+\frac{n\cdot(n+1)}{2}\cdot\lfloor\frac{b}{c}\rfloor+g(a\ mod\ c,b\ mod\ c,c,n)&a\ge{c}\text{ 或 }b\ge{c}\\ \frac{1}{2}\cdot{n}\cdot(n+1)\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor-\frac{1}{2}\cdot{h}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)\\ \ -\frac{1}{2}\cdot{f}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)&a<c\text{ 且 }b<c\\ \end{cases} \]
    \[ h(a,b,c,n)=\sum_{i=0}^n\left\lfloor\frac{a\cdot{i}+b}{c}\right\rfloor^2= \begin{cases} (n+1)\cdot\lfloor\frac{b}{c}\rfloor^2+n\cdot(n+1)\cdot\lfloor\frac{a}{c}\rfloor\cdot\lfloor\frac{b}{c}\rfloor+\frac{n\cdot(n+1)\cdot(2n+1)}{6}\cdot\lfloor\frac{a}{c}\rfloor^2\\ +2\cdot\lfloor\frac{b}{c}\rfloor\cdot{f}(a\ mod\ c,b\ mod\ c,c,n)+2\cdot\lfloor\frac{a}{c}\rfloor\cdot{g}(a\ mod\ c,b\ mod\ c,c,n)&a\ge{c}\text{ 或 }b\ge{c}\\ n\cdot\lfloor\frac{a\cdot{n}+b}{c}\rfloor\cdot(\lfloor\frac{a\cdot{n}+b}{c}\rfloor+1)-2\cdot{g}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)\\ -2\cdot{f}(c,c-b-1,a,\lfloor\frac{a\cdot{n}+b}{c}\rfloor-1)-f(a,b,c,n)&a<c\text{ 且 }b<c\\ \end{cases} \]

    #include <bits/stdc++.h>
using namespace std;
// 2023 OneWan
const long long mod = 998244353;
long long qpow(long long a, long long b) {
    long long res = 1;
    while (b) {
        if (b & 1) {
            res = res * a % mod;
        }
        a = a * a % mod;
        b >>= 1;
    }
    return res;
}
const long long i2 = qpow(2, mod - 2);
const long long i6 = qpow(6, mod - 2);
struct Node {
    long long f, g, h;
    Node() = default;
    Node(long long f, long long g, long long h) : f(f), g(g), h(h) {}
};
Node calc(int n, int a, int b, int c) {
    int ac = a / c, bc = b / c;
    long long m = (1LL * a * n + b) / c;
    int n1 = n + 1, n2 = n << 1 | 1;
    if (a == 0) {
        return Node(1LL * bc * n1 % mod, 1LL * bc * n % mod * n1 % mod * i2 % mod, 1LL * bc * bc % mod * n1 % mod);
    }
    if (a >= c || b >= c) {
        Node res = calc(n, a % c, b % c, c);
        return Node(
            ((1LL * n * n1 % mod * i2 % mod * ac % mod + 1LL * bc * n1 % mod) % mod + res.f) % mod,
            ((1LL * ac * n % mod * n1 % mod * n2 % mod * i6 % mod + 1LL * bc * n % mod * n1 % mod * i2 % mod) % mod + res.g) % mod,
            (((((1LL * ac * ac % mod * n % mod * n1 % mod * n2 % mod * i6 % mod + 1LL * bc * bc % mod * n1 % mod) % mod + 1LL * ac * bc % mod * n % mod * n1 % mod) % mod + res.h) % mod + 2LL * bc % mod * res.f % mod) % mod + 2LL * ac % mod * res.g % mod) % mod);
    }
    Node res = calc(m - 1, c, c - b - 1, a);
    long long f = (1LL * n * m % mod - res.f + mod) % mod;
    return Node(
        f,
        ((1LL * m * n % mod * n1 % mod - res.h - res.f) % mod + mod) * i2 % mod,
        ((1LL * n * m % mod * (m + 1) % mod - 2LL * res.g - 2LL * res.f - f) % mod + mod) % mod);
}
void solve() {
    int n, a, b, c;
    cin >> n >> a >> b >> c;
    auto [f, g, h] = calc(n, a, b, c);
    cout << f << " " << h << " " << g << "\n";
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}

            res = res * a % mod;
        }
        a = a * a % mod;

    int ac = a / c, bc = b / c;
    long long m = (1LL * a * n + b) / c;
    int n1 = n + 1, n2 = n << 1 | 1;
    if (a == 0) {
        return 1LL * bc * n1 % mod;
    }
    if (a >= c || b >= c) {
        long long res = calc(n, a % c, b % c, c);
        return 1LL * n * n1 % mod * i2 % mod * ac % mod + 1LL * bc * n1 % mod) % mod + res) % mod;
    }
    long long res = calc(m - 1, c, c - b - 1, a);
    return (1LL * n * m % mod - res + mod) % mod;
}
