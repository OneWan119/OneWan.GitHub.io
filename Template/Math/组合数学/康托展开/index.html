    算法定义

    求一个 \(1~n\) 的任意排列的排名

    long long cantor(const vector<int> &p, const int n) {
    long long res = 1;
    FenwickTree<int> ft(n + 1);
    for (int i = n - 1 ; i >= 0 ; i--) {
        res = (res + f[n - 1 - i] * ft.query(p[i]) % mod) % mod;
        ft.add(p[i], 1);
    }
    return res;
}

    vector<int> decantor(long long x, const int n) {
    x--;
    Treap<int> tr(n);
    vector<long long> a(n);
    vector<int> b(n);
    for (int i = 1 ; i <= n ; i++) {
        tr.insert(i);
    }
    for (int i = 0 ; i < n ; i++) {
        a[i] = x / f[n - i - 1];
        x %= f[n - i - 1];
    }
    for (int i = 0 ; i < n ; i++) {
        b[i] = tr.queryNum(a[i] + 1);
        tr.erase(b[i]);
    }
    return b;
}

// 2023 OneWan
int lowbit(const int x) {
    return x & -x;
}
template<class T> struct FenwickTree {
    vector<T> sum;
    int size;
    FenwickTree() {}
    FenwickTree(const int n) {
        size = n;
        sum.assign(n + 1, 0);
    }
    void clear() {
        sum.clear();
    }
    T query(int x) {
        T res = 0;
        while (x) {
            res += sum[x];
            x -= lowbit(x);
        }
        return res;
    }
    T query(const int L, const int R) {
        return query(R) - query(L - 1);
    }
    void add(int x, const T k) {
        while (x <= size) {
            sum[x] += k;
            x += lowbit(x);
        }
    }
}; // FenwickTree
template<class T> struct Info {
    int L, R, rnd, cnt, size;
    T val;
    Info() = default;
    Info(const int L, const int R, const int rnd, const int cnt, const int size, const T val) : L(L), R(R), rnd(rnd), cnt(cnt), size(size), val(val) {}
};
template<class T> struct Treap {
    vector<Info<T>> info;
    int tot, root;
    Treap(const int N) : tot(0), root(0), info(N + 1) {}
    void push(const int p) {
        info[p].size = info[info[p].L].size + info[info[p].R].size + info[p].cnt;
    }
    void LRotate(int &p) {
        int temp = info[p].R;
        info[p].R = info[temp].L;
        info[temp].L = p;
        info[temp].size = info[p].size;
        push(p);
        p = temp;
    }
    void RRotate(int &p) {
        int temp = info[p].L;
        info[p].L = info[temp].R;
        info[temp].R = p;
        info[temp].size = info[p].size;
        push(p);
        p = temp;
    }
    void build(int &p, const T x) {
        p = ++tot;
        info[p] = Info(0, 0, rand(), 1, 1, x);
    }
    void insert(const T x) {
        insert(root, x);
    }
    void insert(int &p, const T x) {
        if (p == 0) {
            build(p, x);
            return;
        }
        info[p].size++;
        if (info[p].val == x) {
            info[p].cnt++;
        } else if (info[p].val < x) {
            insert(info[p].R, x);
            if (info[info[p].R].rnd < info[p].rnd) {
                LRotate(p);
            }
        } else {
            insert(info[p].L, x);
            if (info[info[p].L].rnd < info[p].rnd) {
                RRotate(p);
            }
        }
    }
    bool erase(const T x) {
        return erase(root, x);
    }
    bool erase(int &p, const T x) {
        if (p == 0) {
            return false;
        }
        if (info[p].val == x) {
            if (info[p].cnt > 1) {
                info[p].cnt--;
                info[p].size--;
                return true;
            }
            if (info[p].L == 0 || info[p].R == 0) {
                p = info[p].L + info[p].R;
                return true;
            } else if (info[info[p].L].rnd < info[info[p].R].rnd) {
                RRotate(p);
                return erase(p, x);
            } else {
                LRotate(p);
                return erase(p, x);
            }
        } else if (info[p].val < x) {
            bool succ = erase(info[p].R, x);
            if (succ) info[p].size--;
            return succ;
        } else {
            bool succ = erase(info[p].L, x);
            if (succ) info[p].size--;
            return succ;
        }
    }
    int queryRank(const T x) {
        return queryRank(root, x);
    }
    int queryRank(const int p, const T x) {
        if (p == 0) return 0;
        if (info[p].val == x) {
            return info[info[p].L].size + 1;
        } else if (info[p].val < x) {
            return info[info[p].L].size + info[p].cnt + queryRank(info[p].R, x);
        } else {
            return queryRank(info[p].L, x);
        }
    }
    T queryNum(const T x) {
        return queryNum(root, x);
    }
    T queryNum(const int p, const T x) {
        if (p == 0) return 0;
        if (x <= info[info[p].L].size) {
            return queryNum(info[p].L, x);
        } else if (x > info[info[p].L].size + info[p].cnt) {
            return queryNum(info[p].R, x - info[info[p].L].size - info[p].cnt);
        } else {
            return info[p].val;
        }
    }
    int queryPre(const T x) {
        return queryPre(root, x);
    }
    int queryPre(const int p, const T x) {
        if (p == 0) return -1;
        if (info[p].val < x) {
            int res = queryPre(info[p].R, x);
            if (res != -1) return res;
            return p;
        }
        return queryPre(info[p].L, x);
    }
    int querySub(const T x) {
        return querySub(root, x);
    }
    int querySub(const int p, const T x) {
        if (p == 0) return -1;
        if (info[p].val > x) {
            int res = querySub(info[p].L, x);
            if (res != -1) return res;
            return p;
        } else {
            return querySub(info[p].R, x);
        }
    }
    T &operator[](const int p) {
        return info[p].val;
    }
}; // Treap
namespace CantorExpansion {
    const int& mod = 998244353;
    vector<long long> f(1, 1);
    void init(const int n) {
        const int k = f.size();
        if (n > k) {
            f.resize(n + 1);
            for (int i = k ; i <= n ; i++) {
                f[i] = f[i - 1] * i % mod;
            }
        } else return;
    }
    long long cantor(const vector<int>& p, const int n) {
        init(n);
        long long res = 1;
        FenwickTree<int> ft(n);
        for (int i = n - 1 ; i >= 0 ; i--) {
            res = (res + f[n - 1 - i] * ft.query(p[i]) % mod) % mod;
            ft.add(p[i], 1);
        }
        return res;
    }
    vector<int> decantor(long long x, const int n) {
        init(n);
        x--;
        Treap<int> tr(n);
        vector<long long> a(n);
        vector<int> b(n);
        for (int i = 1 ; i <= n ; i++) {
            tr.insert(i);
        }
        for (int i = 0 ; i < n ; i++) {
            a[i] = x / f[n - i - 1];
            x %= f[n - i - 1];
        }
        for (int i = 0 ; i < n ; i++) {
            b[i] = tr.queryNum(a[i] + 1);
            tr.erase(b[i]);
        }
        return b;
    }
} // CantorExpansion
