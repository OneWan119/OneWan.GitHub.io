    #include <bits/stdc++.h>
using namespace std;
// 2023 OneWan
long long dp[20];
int num[20];
long long dfs(int x, bool lim) {
    if (x == -1) {
        return 1LL;
    }
    if (lim == false && dp[x] != -1) {
        return dp[x];
    }
    int mx = lim ? num[x] : 9;
    long long res = 0;
    for (int i = 0 ; i <= mx ; i++) {
        if (i == 4) {
            continue;
        }
        res += dfs(x - 1, lim && i == mx);
    }
    if (lim == false) {
        dp[x] = res;
    }
    return res;
}
long long get(long long x) {
    int cnt = 0;
    while (x) {
        dp[cnt] = -1;
        num[cnt++] = x % 10;
        x /= 10;
    }
    return dfs(cnt - 1, true);
}
void solve() {
    long long k;
    cin >> k;
    auto check = [&](long long mid) -> bool {
        return get(mid) >= k + 1;
    };
    long long L = k, R = 1e18;
    while (L < R) {
        long long mid = L + R >> 1;
        if (check(mid)) {
            R = mid;
        } else {
            L = mid + 1;
        }
    }
    cout << L << "\n";
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int T;
    cin >> T;
    while (T--) {
        solve();
    }
    return 0;
}

    #include <bits/stdc++.h>
using namespace std;
// 2023 OneWan
long long dp[10][8];
int num[8];
long long dfs(int x, int last, bool lim) {
    if (x == -1) {
        return 1LL;
    }
    if (lim == false && dp[last][x] != -1) {
        return dp[last][x];
    }
    int mx = lim ? num[x] : 9;
    long long res = 0;
    for (int i = 0 ; i <= mx ; i++) {
        if ((last == 6 && i == 2) || i == 4) {
            continue;
        }
        res += dfs(x - 1, i, lim && i == mx);
    }
    if (lim == false) {
        dp[last][x] = res;
    }
    return res;
}
long long get(long long x) {
    int cnt = 0;
    while (x) {
        for (int i = 0 ; i < 10 ; i++) {
            dp[i][cnt] = -1;
        }
        num[cnt++] = x % 10;
        x /= 10;
    }
    return dfs(cnt - 1, 0, true);
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, m;
    while (cin >> n >> m) {
        if (n == 0) break;
        cout << get(m) - get(n - 1) << "\n";
    }
    return 0;
}

    #include <bits/stdc++.h>
using namespace std;
// 2023 OneWan
long long dp[20][2][20][20][20];
int num[20];
long long dfs(int x, bool has, int len, int cnt5, int cnt7, bool lim) {
    has |= len >= 7;
    if (x == -1) {
        return cnt5 + 3 * cnt7 + 300 * has;
    }
    if (lim == false && dp[x][has][len][cnt5][cnt7] != -1) {
        return dp[x][has][len][cnt5][cnt7];
    }
    int mx = lim ? num[x] : 9;
    long long res = 0;
    for (int i = 0 ; i <= mx ; i++) {
        int rlen = (i == 7 ? len + 1 : 0);
        res += dfs(x - 1, has, rlen, cnt5 + (i == 5), cnt7 + (i == 7), lim && i == mx);
    }
    if (lim == false) {
        dp[x][has][len][cnt5][cnt7] = res;
    }
    return res;
}
long long get(long long x) {
    int cnt = 0;
    while (x) {
        num[cnt++] = x % 10;
        x /= 10;
    }
    return dfs(cnt - 1, false, 0, 0, 0, true);
}
void solve() {
    long long L, R;
    cin >> L >> R;
    cout << get(R) - get(L - 1) << "\n";
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    memset(dp, -1, sizeof dp);
    int t;
    cin >> t;
    while (t--) {
        solve();
    }
    return 0;
}
