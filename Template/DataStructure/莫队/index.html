    #include <bits/stdc++.h>
using namespace std;
// 2023 OneWan
int belong[1000005];
struct MQuery {
    int L, R, idx;
    bool operator<(const MQuery& x) const {
        if (belong[L] != belong[x.L]) return L < x.L;
        if (belong[L] & 1) return R < x.R;
        return R > x.R;
    }
};
int C[50005], cnt[50005];
long long sum;
void add(int x) {
    sum += cnt[x];
    cnt[x]++;
}
void del(int x) {
    cnt[x]--;
    sum -= cnt[x];
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int N, M;
    cin >> N >> M;
    int mx = sqrt(N);
    for (int i = 1 ; i <= N ; i++) {
        belong[i] = i / mx;
    }
    for (int i = 1 ; i <= N ; i++) {
        cin >> C[i];
    }
    vector<MQuery> mq(M);
    for (int i = 0 ; i < M ; i++) {
        cin >> mq[i].L >> mq[i].R;
        mq[i].idx = i;
    }
    sort(begin(mq), end(mq));
    vector<pair<long long, long long>> ans(M);
    for (int i = 0, L = 1, R = 0 ; i < M ; i++) {
        if (mq[i].L == mq[i].R) {
            ans[mq[i].idx] = {0, 1};
            continue;
        }
        while (L > mq[i].L) add(C[--L]);
        while (R < mq[i].R) add(C[++R]);
        while (L < mq[i].L) del(C[L++]);
        while (R > mq[i].R) del(C[R--]);
        ans[mq[i].idx] = {sum, 1LL * (R - L + 1) * (R - L) / 2};
    }
    for (int i = 0 ; i < M ; i++) {
        auto [p, q] = ans[i];
        long long d = __gcd(p, q);
        p /= d;
        q /= d;
        cout << p << "/" << q << "\n";
    }
    return 0;
}

using namespace std;

int belong[1000005];
struct MQuery {
    int L, R, idx, time;        
    bool operator<(const MQuery& x) const {
        if (belong[L] != belong[x.L]) return L < x.L;
        if (belong[R] == belong[x.R]) return time < x.time;
        if (belong[L] & 1) return R < x.R;
        return R > x.R;
    }
};
int color[1000005], cnt[1000005];
vector<MQuery> mq;
vector<pair<int, int>> change(1);
int res;
void add(int x) {
    if (cnt[x]++ == 0) res++;
}
void del(int x) {
    if (cnt[x]-- == 1) res--;
}
void modify(int now, int i) {
    if (change[now].first >= mq[i].L && change[now].first <= mq[i].R) {
        del(color[change[now].first]);
        add(change[now].second);
    }
    swap(color[change[now].first], change[now].second);
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int N, M;
    cin >> N >> M;
    int mx = pow(N, 0.6666666666);
    for (int i = 1 ; i <= N ; i++) {
        belong[i] = i / mx;
    }
    for (int i = 1 ; i <= N ; i++) {
        cin >> color[i];
    }
    int qcnt = 0, ccnt = 0;
    for (int i = 0 ; i < M ; i++) {
        char ch;
        int a, b;
        cin >> ch >> a >> b;
        if (ch == 'Q') {
            mq.push_back({a, b, qcnt, ccnt});
            qcnt++;
        } else {
            change.push_back({a, b});
            ccnt++;
        }
    }
    vector<int> ans(qcnt);
    sort(begin(mq), end(mq));
    for (int i = 0, L = 1, R = 0, now = 0 ; i < qcnt ; i++) {
        while (L > mq[i].L) add(color[--L]);
        while (R < mq[i].R) add(color[++R]);
        while (L < mq[i].L) del(color[L++]);
        while (R > mq[i].R) del(color[R--]);
        while (now > mq[i].time) modify(now--, i);
        while (now < mq[i].time) modify(++now, i);
        ans[mq[i].idx] = res;
    }
    for (int i = 0 ; i < qcnt ; i++) {
        cout << ans[i] << "\n";
    }
    return 0;
}

int belong[100005], a[100005], cnt[200005], mex[100005];
struct MQuery {
    int L, R, idx, time;        
    bool operator<(const MQuery& x) const {
        if (belong[L] != belong[x.L]) return L < x.L;
        if (belong[R] == belong[x.R]) return time < x.time;
        if (belong[L] & 1) return R < x.R;
        return R > x.R;
    }
};
vector<MQuery> mq;
vector<pair<int, int>> change(1);
void add(int x) {
    mex[cnt[x]]--;
    cnt[x]++;
    mex[cnt[x]]++;
}
void del(int x) {
    mex[cnt[x]]--;
    cnt[x]--;
    mex[cnt[x]]++;
}
void modify(int now, int i) {
    if (change[now].first >= mq[i].L && change[now].first <= mq[i].R) {
        del(a[change[now].first]);
        add(change[now].second);
    }
    swap(a[change[now].first], change[now].second);
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int n, q;
    cin >> n >> q;
    int mx = pow(n, 0.6666666666);
    for (int i = 1 ; i <= n ; i++) {
        belong[i] = i / mx;
    }
    vector<int> temp(1);
    for (int i = 1 ; i <= n ; i++) {
        cin >> a[i];
        temp.emplace_back(a[i]);
    }
    int qcnt = 0, ccnt = 0;
    for (int i = 0 ; i < q ; i++) {
        int op, L, R;
        cin >> op >> L >> R;
        if (op == 1) {
            mq.push_back({L, R, qcnt, ccnt});
            qcnt++;
        } else {
            change.emplace_back(L, R);
            temp.emplace_back(R);
            ccnt++;
        }
    }
    sort(begin(mq), end(mq));
    sort(begin(temp), end(temp));
    temp.resize(unique(begin(temp), end(temp)) - begin(temp));
    for (int i = 1 ; i <= n ; i++) {
        a[i] = lower_bound(begin(temp), end(temp), a[i]) - begin(temp);
    }
    for (auto &[x, y] : change) {
        y = lower_bound(begin(temp), end(temp), y) - begin(temp);
    }
    vector<int> ans(qcnt);
    for (int i = 0, L = 1, R = 0, now = 0 ; i < qcnt ; i++) {
        while (L > mq[i].L) add(a[--L]);
        while (R < mq[i].R) add(a[++R]);
        while (L < mq[i].L) del(a[L++]);
        while (R > mq[i].R) del(a[R--]);
        while (now > mq[i].time) modify(now--, i);
        while (now < mq[i].time) modify(++now, i);
        int res = 1;
        while (mex[res]) res++;
        ans[mq[i].idx] = res;
    }
    for (int i = 0 ; i < qcnt ; i++) {
        cout << ans[i] << "\n";
    }
    return 0;
}

    #include <bits/stdc++.h>
using namespace std;
// 2023 OneWan
int belong[200005];
struct MQuery {
    int L, R, idx;
    bool operator<(const MQuery& x) const {
        if (belong[L] != belong[x.L]) return L < x.L;
        return R < x.R;
    }
};
int a[200005], b[200005];
pair<int, int> cnt[200005], cnt1[200005];
int res;
void addRight(int i) {
    if (cnt[a[i]].first == 0) cnt[a[i]].first = i;
    cnt[a[i]].second = i;
    res = max(res, cnt[a[i]].second - cnt[a[i]].first);
}
void addLeft(int i) {
    if (cnt1[a[i]].second == 0) cnt1[a[i]].second = i;
    cnt1[a[i]].first = i;
    res = max(res, max(cnt1[a[i]].second, cnt[a[i]].second) - cnt1[a[i]].first);
}
void del(int i) {
    cnt[a[i]] = {0, 0};
}
void rollBack(int i) {
    cnt1[a[i]] = {0, 0};
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int N;
    cin >> N;
    int mx = sqrt(N);
    for (int i = 1 ; i <= N ; i++) {
        belong[i] = (i - 1)/ mx + 1;
    }
    for (int i = 1 ; i <= belong[N] ; i++) {
        b[i] = (i == belong[N]) ? N : mx * i;
    }
    vector<int> c(1);
    for (int i = 1 ; i <= N ; i++) {
        cin >> a[i];
        c.emplace_back(a[i]);
    }
    sort(begin(c), end(c));
    c.resize(unique(begin(c), end(c)) - begin(c));
    for (int i = 1 ; i <= N ; i++) {
        a[i] = lower_bound(begin(c), end(c), a[i]) - begin(c);
    }
    int M;
    cin >> M;
    vector<MQuery> mq;
    for (int i = 0 ; i < M ; i++) {
        int L, R;
        cin >> L >> R;
        mq.emplace_back(L, R, i);
    }
    sort(begin(mq), end(mq));
    vector<int> ans(M);
    int last = belong[mq[0].L], R = b[last], L = R + 1;
    for (int i = 0 ; i < M ; i++) {
        if (belong[mq[i].L] == belong[mq[i].R]) {
            for (int j = mq[i].L ; j <= mq[i].R ; j++) {
                if (cnt1[a[j]].first == 0) cnt1[a[j]].first = j;
                cnt1[a[j]].second = j;
            }
            int temp = 0;
            for (int j = mq[i].L ; j <= mq[i].R ; j++) {
                temp = max(temp, cnt1[a[j]].second - cnt1[a[j]].first);
            }
            for (int j = mq[i].L ; j <= mq[i].R ; j++) {
                cnt1[a[j]] = {0, 0};
            }
            ans[mq[i].idx] = temp;
            continue;
        }
        if (last != belong[mq[i].L]) {
            while (R > b[belong[mq[i].L]]) del(R--);
            while (L < b[belong[mq[i].L]] + 1) del(L++);
            res = 0;
            last = belong[mq[i].L];
        }
        while (R < mq[i].R) addRight(++R);
        int temp = res, L1 = L;
        while (L1 > mq[i].L) addLeft(--L1);
        ans[mq[i].idx] = res;
        while (L1 < L) rollBack(L1++);
        res = temp;
    }
    for (int i = 0 ; i < M ; i++) {
        cout << ans[i] << "\n";
    }
    return 0;
}

struct MQuery {

        if (belong[L] != belong[x.L]) return L < x.L;
        return R > x.R;
    }
};
int a[200005], b[200005];
int cnt[200005], cnt1[200005];
int res;
void add(int i) {
    cnt[a[i]]++;
}
void del(int i) {
    if (cnt[a[i]]-- == 1) {
        res = min(res, a[i]);
    }
}
int main() {
    ios::sync_with_stdio(0);
    cin.tie(0);
    cout.tie(0);
    int N, M;
    cin >> N >> M;
    int mx = sqrt(N);
    for (int i = 1 ; i <= N ; i++) {
        belong[i] = (i - 1)/ mx + 1;

    for (int i = 1 ; i <= belong[N] ; i++) {
        b[i] = mx * (i - 1) + 1;

            for (int j = mq[i].L ; j <= mq[i].R ; j++) {
                cnt1[a[j]]++;
            }
            int temp = 0;
            while (cnt1[temp]) temp++;
            for (int j = mq[i].L ; j <= mq[i].R ; j++) {
                cnt1[a[j]]--;
            }
            ans[mq[i].idx] = temp;
            continue;

        if (belong[mq[i].L] != last) {
            while (R < N) add(++R);
            while (L < b[belong[mq[i].L]]) del(L++);
            int temp = 0;
            while (cnt[temp]) temp++;
            res = temp;
            last = belong[mq[i].L];

        while (R > mq[i].R) del(R--);

        while (L1 > L) add(--L1);
        res = temp;
    }
    for (int i = 0 ; i < M ; i++) {
        cout << ans[i] << "\n";
    }
