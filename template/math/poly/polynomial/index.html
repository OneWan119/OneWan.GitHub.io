using Poly = vector<int>;
const int P = 998244353;
#define mod_mul(a, b) (a = i64(a) * (b) % P)
#define mod_mul_t(a, b) (i64(a) * (b) % P)
#define mod_add(a, b) (((a) += (b)) >= P ? (a) -= P : 0)
#define mod_add_t(a, b) ((a) >= P - (b) ? (a) + (b) - P : (a) + (b))
#define mod_sub(a, b) (((a) -= (b)) < 0 ? (a) += P : 0)
#define mod_sub_t(a, b) ((a) < (b) ? (a) - (b) + P : (a) - (b))
i64 qpow(i64 a, int b = P - 2, i64 res = 1) {
    while (b) {
        if (b & 1) res = res * a % P;
        a = a * a % P;
        b >>= 1;
    }
    return res;
}

    void fft(vector<Complex> &a) {
        if (r.empty()) {
            r = [](int len) {
                vector<Complex> r(1 << len);
                for (int i = 0 ; i < len ; i++) {
                    int L = 1 << i;
                    r[L] = Complex(cos(PI2 / L), sin(PI2 / L));
                    for (int j = L + 1 ; j < (L << 1) ; j++) {
                        r[j] = r[j - L] * r[L];
                    }
                }
                return r;
            }(20);
        }
        int len = a.size();
        for (int i = len ; i >= 2 ; i /= 2) {
            int L = i / 2;
            for (int j = 0 ; j < L ; j++) {
                Complex z = a[j + L];
                a[j + L] = a[j] - z;
                a[j] = a[j] + z;
            }
            for (int j = i, res = 1 ; j < len ; j += i, res++) {
                for (int k = 0 ; k < L ; k++) {
                    Complex z = a[j + k + L] * r[res];
                    a[j + k + L] = a[j + k] - z;
                    a[j + k] = a[j + k] + z;
                }
            }
        }
    }
    void ifft(vector<Complex> &a) {
        int len = a.size();
        for (int i = 2 ; i <= len ; i *= 2) {
            int L = i / 2;
            for (int j = 0 ; j < L ; j++) {
                Complex z = a[j + L];
                a[j + L] = a[j] - z;
                a[j] = a[j] + z;
            }
            for (int j = i, res = 1 ; j < len ; j += i, res++) {
                for (int k = 0 ; k < L ; k++) {
                    Complex z = a[j + k + L];
                    a[j + k + L] = (a[j + k] - z) * r[res];
                    a[j + k] = a[j + k] + z;
                }
            }
        }
        for (int i = 0 ; i < len ; i++) {
            a[i] = a[i] / len;
        }
        reverse(begin(a) + 1, end(a));
    }

    const int g = 3; // P 的原根
    const int I = 86583718; // g^((P-1)/4)
    // NTT 模数 mod 必须为 p*2^(q)+1 (p,q为任意正整数)
    // 167772161 | 3
    // 469762049 | 3
    // 754974721 | 11
    // 998244353 | 3
    // 1004535809 | 3
    Poly W;
    template<class T>
    void DIF(T *a, int n) {
        if (W.empty()) {
            W = [](int L) {
                int wn = qpow(g, P / L);
                Poly w(L);
                w[L >> 1] = 1;
                for (int i = L / 2 + 1 ; i < L ; i++) {
                    w[i] = mod_mul_t(w[i - 1], wn);
                }
                for (int i = L / 2 - 1 ; i >= 1 ; i--) {
                    w[i] = w[i << 1];
                }
                return w;
            }(1 << 20);
        }
        for (int k = n >> 1 ; k ; k >>= 1) {
            for (int i = 0 ; i < n ; i += k << 1) {
                i64 x, y;
                for (int j = 0 ; j < k ; j++) {
                    x = a[i + j]; y = a[i + j + k];
                    a[i + j] = mod_add_t(a[i + j], y);
                    a[i + j + k] = mod_mul_t(mod_sub_t(x, y), W[j + k]);
                }
            }
        }
    }
    template<class T>
    void IDIT(T *a, int n) {
        for (int k = 1 ; k < n ; k <<= 1) {
            for (int i = 0 ; i < n ; i += k << 1) {
                i64 x, y;
                for (int j = 0 ; j < k ; j++) {
                    x = a[i + j]; y = mod_mul_t(a[i + j + k], W[j + k]);
                    a[i + j] = mod_add_t(x, y);
                    a[i + j + k] = mod_sub_t(x, y);
                }
            }
        }
        int inv = P - (P - 1) / n;
        for (int i = 0 ; i < n ; i++) {
            mod_mul(a[i], inv);
        }
        reverse(a + 1, a + n);
    }

            Poly c(n);
            for (int i = 0 ; i < (int) a.size() ; i++) {
                for (int j =  0; j < (int) b.size() ; j++) {
                    mod_add(c[i + j], mod_mul_t(a[i], b[j]));
                }
            }
            return c;
        }
        int L = norm(n);
        a.resize(L); b.resize(L);
        DFT(a); DFT(b); dot(a, b); IDFT(a);
        return a.resize(n), a;
    }

    Poly mtt_f(const Poly &f, const Poly &g, int mod) {
        int n = f.size() + g.size() - 1;
        if (f.size() <= 16 || g.size() <= 16) {
            Poly c(n);
            for (int i = 0 ; i < (int) f.size() ; i++) {
                for (int j =  0; j < (int) g.size() ; j++) {
                    c[i + j] = (c[i + j] + (i64) f[i] * g[j] % mod) % mod;
                }
            }
            return c;
        }
        int L = norm(n);
        vector<FFT::Complex> a(L), b(L), c(L);
        for (int i = 0 ; i < (int) f.size() ; i++) {
            a[i] = {f[i] & 65535, f[i] >> 16};
        }
        for (int i = 0 ; i < (int) g.size() ; i++) {
            b[i] = {g[i] & 65535, g[i] >> 16};
        }
        FFT::fft(a);
        FFT::fft(b);
        c = a;
        a[0] = a[0] * (b[0].x * 2);
        c[0] = c[0] * (b[0].y * 2);
        int w = FFT::binary_upper_bound_in_bits(L - 1);
        for (int k = 0 ; k < w ; k++) {
            int l = 1 << k, mask = l - 1;
            for (int i = l ; i < (l << 1) ; i++) {
                FFT::Complex p = b[i], q = b[i ^ mask].conj(), x = (p + q), y = (q - p) * FFT::I;
                a[i] = a[i] * x;
                c[i] = c[i] * y;
            }
        }
        FFT::ifft(a);
        FFT::ifft(c);
        Poly res(n);
        auto get = [&](double x) {
            i64 v = x + .5;
            return ((v < 0) ? (mod + (v - 1) / 2 % mod) : (v + 1) / 2) % mod;
        };
        for (int i = 0 ; i < n ; i++) {
            res[i] = (
                (get(c[i].y) << (16 << 1)) +
                (get(c[i].x + a[i].y) << 16) +
                get(a[i].x)
            ) % mod;
        }
        return res;
    }
    Poly inv2k(Poly a) {
        int n = a.size(), m = n >> 1;;
        if (n == 1) {
            return Poly(1, qpow(a[0]));
        }
        Poly b = inv2k(Poly(begin(a), begin(a) + m));
        Poly c = b;
        b.resize(n); DFT(a); DFT(b); dot(a, b); IDFT(a);
        for (int i = 0 ; i < n ; i++) {
            a[i] = (i < m ? 0 : P - a[i]);
        }
        DFT(a); dot(a, b); IDFT(a);
        return move(begin(c), end(c), begin(a)), a;
    }

    Poly Exp(Poly a) {
        assert(a[0] == 0);
        int n = a.size(), k = norm(n);
        Poly b = {1}, c, d;
        a.resize(k);
        for (int L = 2 ; L <= k ; L <<= 1) {
            d = b; b.resize(L);
            c = Ln(b); c.resize(L);
            for (int i = 0 ; i < L ; i++) {
                c[i] = mod_sub_t(a[i], c[i]);
            }
            mod_add(c[0], 1); DFT(b); DFT(c); dot(b, c); IDFT(b);
            move(begin(d), end(d), begin(b));
        }
        return b.resize(n), b;
    }
    void Pow(Poly &a, int m0, int m1, int d) {
        int n = a.size();
        a.erase(begin(a), begin(a) + d);
        int a0 = a[0]; assert(a0);
        a *= qpow(a0);
        a = Exp(Ln(a) * m0) * qpow(a0, m1);
        a.resize(n); d *= m0;
        for (int i = n - 1 ; i >= 0 ; i--) {
            a[i] = (i >= d ? a[i - d] : 0);
        }
    }
    Poly Pow(Poly a, int m) {
        int n = a.size(), d = 0;
        while (d < n && a[d] == 0) {
            d++;
        }
        if (d == n) return a;
        if ((i64) d * m >= n) {
            return Poly(n);
        }
        Pow(a, m, m, d);
        return a;
    }
    Poly Pow(Poly a, string str) {
        int n = a.size(), d = 0;
        while (d < n && a[d] == 0) {
            d++;
        }
        if (d == n) return a;
        if (d) {
            if (str.size() > 8 || d * stoll(str) >= n) {
                return Poly(n);
            }
        }
        i64 m0 = 0, m1 = 0;
        const int phi = P - 1;
        for (int i = 0 ; i < (int) str.size() ; i++) {
            m0 = (m0 * 10 + str[i] - '0') % P;
            m1 = (m1 * 10 + str[i] - '0') % phi;
        }
        Pow(a, m0, m1, d);
        return a;
    }
    pair<Poly, Poly> Div(Poly f, Poly g) {
        int n = f.size() - 1, m = g.size() - 1;
        if (n < m) {
            return {Poly(1), f};
        }
        Poly f_rev = ~f, g_rev = ~g;
        g_rev.resize(n - m + 1);
        Poly q = f_rev * Inv(g_rev);
        q.resize(n - m + 1); q = ~q;
        g *= q;
        f.resize(m); g.resize(m);
        Poly r = f - g;
        return {q, r};
    }

    Poly Sqrt(Poly a) {
        assert(a[0] == 1); // 多项式常数项必须为1
        int n = a.size();
        Poly res{1};
        for (int len = 1 ; len < (n << 1) ; len <<= 1) {
            Poly b = a; b.resize(len);
            res.resize(len);
            Poly c = b * Inv(res); c.resize(len);
            res += c;
            res /= 2;
        }
        return res.resize(n), res;
    }

    Poly Stirling2ndRow(int n) {
        vector<int> fact(n + 1), invfact(n + 1);
        fact[0] = 1;
        for (int i = 1 ; i <= n ; i++) {
            fact[i] = mod_mul_t(fact[i - 1], i);
        }
        invfact[n] = qpow(fact[n]);
        for (int i = n ; i >= 1 ; i--) {
            invfact[i - 1] = mod_mul_t(invfact[i], i);
        }
        Poly a(n + 1), b(n + 1);
        for (int i = 0 ; i <= n ; i++) {
            a[i] = mod_mul_t(invfact[i], qpow(i, n));
            b[i] = mod_mul_t(invfact[i], (i & 1 ? P - 1 : 1));
        }
        a *= b;
        return a.resize(n + 1), a;
    }

    Poly Stirling1stRow(int n) {
        vector<int> fact(n + 1), invfact(n + 1);
        fact[0] = 1;
        for (int i = 1 ; i <= n ; i++) {
            fact[i] = mod_mul_t(fact[i - 1], i);
        }
        invfact[n] = qpow(fact[n]);
        for (int i = n ; i >= 1 ; i--) {
            invfact[i - 1] = mod_mul_t(invfact[i], i);
        }
        function<Poly(int)> dfs = [&](int n) -> Poly {
            if (n == 1) {
                return {0, 1};
            }
            if (n & 1) {
                Poly a = dfs(n - 1);
                a.resize(n + 1);
                for (int i = n ; i > 0 ; i--) {
                    mod_mul(a[i], n - 1);
                    mod_add(a[i], a[i - 1]);
                }
                mod_mul(a[0], n - 1);
                return a;
            } else {
                Poly a = dfs(n / 2);
                int len = n / 2;
                a.resize(len + 1);
                Poly b(len + 1), c(len + 1);
                for (int i = 0 ; i <= len ; i++) {
                    b[i] = mod_mul_t(a[len - i], fact[len - i]);
                    c[i] = mod_mul_t(qpow(len, i), invfact[i]);
                }
                b *= c;
                for (int i = 0 ; i <= len ; i++) {
                    c[i] = mod_mul_t(b[len - i], invfact[i]);
                }
                a *= c;
                return a;
            }
        };
        return dfs(n);
    }

    Poly BerlekampMassey(const Poly &a) {
        int n = a.size();
        Poly temp(n + 1), f(n + 1), g(n + 1);
        int k = 0, last_k = 0, last_delta, last = -1;
        for (int i = 0 ; i < n ; i++) {
            int delta = P - a[i];
            for (int j = 1 ; j <= k ; j++) {
                mod_add(delta, mod_mul_t(f[j], a[i - j]));
            }
            if (delta == 0) continue;
            if (last == -1) {
                k = i + 1;
            } else {
                int t = mod_mul_t(delta, qpow(last_delta));
                mod_add(temp[i - last], t);
                for (int j = 1 ; j <= last_k ; j++) {
                    mod_sub(temp[i - last + j], mod_mul_t(t, g[j]));
                }
                int p = last_k;
                last_k = k;
                k = max(k, i - last + p);
                for (int j = 1 ; j <= last_k ; j++) {
                    g[j] = f[j];
                }
                for (int j = 1 ; j <= k ; j++) {
                    f[j] = temp[j];
                }
            }
            last_delta = delta;
            last = i;
        }
        return f.resize(k + 1), f;
    }
    i64 LinearRec(const Poly &a, const Poly &r, i64 n) {
        Poly F(begin(a), begin(a) + r.size() - 1), R = -r;
        R[0] = 1;
        F *= R;
        F.resize(R.size() - 1);
        for ( ; n ; n /= 2) {
            Poly rR = R;
            for (int i = 0 ; i < (int) rR.size() ; i += 2) {
                rR[i] = mod_sub_t(P, rR[i]);
            }
            Poly FR = F * rR, RR = R * rR;
            F.resize(0); R.resize(0);
            for (int i = (n & 1) ; i < (int) FR.size() ; i+= 2) {
                F.push_back(FR[i]);
            }
            for (int i = 0 ; i < (int) RR.size() ; i += 2) {
                R.push_back(RR[i]);
            }
        }
        return F.empty() ? 0 : mod_mul_t(F[0],qpow(R[0]));
    }
